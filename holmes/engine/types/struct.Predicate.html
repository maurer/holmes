A Predicate is a name combined with a list of typed slots, e.g.

foo(uint64, string)

would be represented as

use holmes::pg::dyn::types;
use holmes::engine::types::{Predicate, Field};
use std::sync::Arc;
Predicate {
    name: "foo".to_string(),
    description: None,
    fields: vec![Field {
        name: None,
        description: None,
        type_: Arc::new(types::UInt64)
    }, Field {
        name: None,
        description: None,
        type_: Arc::new(types::String)
    }]
};
